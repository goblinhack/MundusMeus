    git clone git@github.com:goblinhack/MundusMeus.git

    Linux:
        ./RUNME

    MacOS:
        ./RUNME

    MacOS+Xcode: (future)
        ./RUNME.Xcode-OSX

    Windows+MSYS2: (future)
        ./RUNME.windows.msys2
